16 static inline bool validate_chronological_times(

18 const Unique_X509_TIME& time_after,
19 const std::optional<uint32_t>& allowed_diff_days = std::nullopt)
20 {

23 CHECK1(ASN1_TIME_diff(&diff_days, &diff_secs, time_before, time_after));
