18 /* Receipts included in snapshots always contain leaf components,
19 including a claims digest and commit evidence, from 2.0.0-rc0 onwards.
20 This verification code deliberately does not support snapshots
21 produced by 2.0.0-dev* releases
22 */
23 static crypto::Sha256Hash compute_root_from_snapshot_receipt(
24 const Receipt& receipt)
25 {
26 crypto::Sha256Hash current;
27 if (receipt.leaf_components.has_value())
28 {
29 auto components = receipt.leaf_components.value();
30 if (
31 components.write_set_digest.has_value() &&
32 components.commit_evidence.has_value() &&
33 components.claims_digest.has_value())
34 {
35 auto ws_dgst = crypto::Sha256Hash::from_hex_string(
36 components.write_set_digest.value());
37 crypto::Sha256Hash ce_dgst(components.commit_evidence.value());
38 auto cl_dgst =
39 crypto::Sha256Hash::from_hex_string(components.claims_digest.value());
40 current = crypto::Sha256Hash(ws_dgst, ce_dgst, cl_dgst);
41 }
42 else
43 {
44 throw std::logic_error(
45 "Cannot compute leaf unless write_set_digest, commit_evidence and "
46 "claims_digest "
47 "are set");
48 }
49 }
50 else
51 {
52 throw std::logic_error(
53 "Cannot compute root if leaf_components are not set");
54 }
55 for (auto const& element : receipt.proof)
56 {
57 if (element.left.has_value())
58 {
59 assert(!element.right.has_value());
60 auto left = crypto::Sha256Hash::from_hex_string(element.left.value());
61 current = crypto::Sha256Hash(left, current);
62 }
63 else
64 {
65 assert(element.right.has_value());
66 auto right = crypto::Sha256Hash::from_hex_string(element.right.value());
67 current = crypto::Sha256Hash(current, right);
68 }
69 }
70
71 return current;
72 }

80 // Store used to verify a snapshot (either created fresh when a node joins

115 static void deserialise_snapshot(
116 const std::shared_ptr<kv::Store>& store,
117 const std::vector<uint8_t>& snapshot,
118 kv::ConsensusHookPtrs& hooks,
119 std::vector<kv::Version>* view_history = nullptr,
120 bool public_only = false,
121 std::optional<kv::Version> evidence_seqno = std::nullopt,
122 std::optional<std::vector<uint8_t>> prev_service_identity = std::nullopt)
123 {
124 const auto* data = snapshot.data();
125 auto size = snapshot.size();
126
127 auto tx_hdr = serialized::peek<kv::SerialisedEntryHeader>(data, size);
128 auto store_snapshot_size = sizeof(kv::SerialisedEntryHeader) + tx_hdr.size;
129
130 // Snapshots without a snapshot evidence seqno specified by the host should
131 // be self-verifiable with embedded receipt
132 if (!evidence_seqno.has_value())
133 {
134 auto receipt_data = data + store_snapshot_size;
135 auto receipt_size = size - store_snapshot_size;
136
137 auto j = nlohmann::json::parse(receipt_data, receipt_data + receipt_size);
138 auto receipt = j.get<Receipt>();
139
140 if (
141 !receipt.leaf_components.has_value() ||
142 !receipt.leaf_components->claims_digest.has_value())
143 {
144 throw std::logic_error(
145 "Snapshot receipt is missing snapshot digest claim");
146 }
147
148 auto snapshot_digest =
149 crypto::Sha256Hash({snapshot.data(), store_snapshot_size});
150 auto snapshot_digest_claim = crypto::Sha256Hash::from_hex_string(
151 receipt.leaf_components->claims_digest.value());
152 if (snapshot_digest != snapshot_digest_claim)
153 {
154 throw std::logic_error(fmt::format(
155 "Snapshot digest ({}) does not match receipt claim ({})",
156 snapshot_digest,
157 snapshot_digest_claim));
158 }
159
160 auto root = compute_root_from_snapshot_receipt(receipt);
161 auto raw_sig = crypto::raw_from_b64(receipt.signature);
162
163 if (!receipt.cert.has_value())
164 {
165 throw std::logic_error("Missing node certificate in snapshot receipt");
166 }
167
168 auto v = crypto::make_unique_verifier(receipt.cert.value());
169 if (!v->verify_hash(
170 root.h.data(), root.h.size(), raw_sig.data(), raw_sig.size()))
171 {
172 throw std::logic_error(
173 "Signature verification failed for snapshot receipt");
174 }
175
176 if (prev_service_identity)
177 {
178 crypto::Pem prev_pem(*prev_service_identity);
179 if (!v->verify_certificate({&prev_pem}, {}, /* ignore_time */ true))
180 {
181 throw std::logic_error(
182 "Previous service identity does not endorse the node identity that "
183 "signed the snapshot");
184 }
185 LOG_DEBUG_FMT("Previous service identity endorses snapshot signer");
186 }
187 }
188
189 LOG_INFO_FMT(
190 "Deserialising snapshot (size: {}, public only: {})",
191 snapshot.size(),
192 public_only);
193
194 auto rc = store->deserialise_snapshot(
195 snapshot.data(), store_snapshot_size, hooks, view_history, public_only);
196 if (rc != kv::ApplyResult::PASS)
197 {
198 throw std::logic_error(fmt::format("Failed to apply snapshot: {}", rc));
199 }
200
201 LOG_INFO_FMT(
202 "Snapshot successfully deserialised at seqno {}",
203 store->current_version());
204 };

206 static std::unique_ptr<StartupSnapshotInfo> initialise_from_snapshot(
207 const std::shared_ptr<kv::Store>& store,
208 std::vector<uint8_t>&& snapshot,

228 static std::vector<uint8_t> build_and_serialise_receipt(
229 const std::vector<uint8_t>& sig,
230 const std::vector<uint8_t>& tree,
231 const NodeId& node_id,
232 const crypto::Pem& node_cert,
233 kv::Version seqno,
234 const crypto::Sha256Hash& write_set_digest,
235 const std::string& commit_evidence,
236 crypto::Sha256Hash&& claims_digest)
237 {
238 ccf::MerkleTreeHistory history(tree);
239 auto proof = history.get_proof(seqno);
240 ccf::ClaimsDigest cd;
241 cd.set(std::move(claims_digest));
242 auto tx_receipt = std::make_shared<ccf::TxReceipt>(
243 sig,
244 proof.get_root(),
245 proof.get_path(),
246 node_id,
247 node_cert,
248 write_set_digest,
249 commit_evidence,
250 cd);
251
252 Receipt receipt = ccf::describe_receipt(tx_receipt);
253 const auto receipt_str = nlohmann::json(receipt).dump();
254 return std::vector<uint8_t>(receipt_str.begin(), receipt_str.end());
255 }
